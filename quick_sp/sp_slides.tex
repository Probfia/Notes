\begin{enumerate}

\item $$\pfrac{U}{V}{T} = T \pfrac{P}{T}{V} - P$$
\item $$\pfrac{U}{P}{T} = -T \pfrac{V}{T}{P} - P \pfrac{V}{P}{T}$$
\end{enumerate}
